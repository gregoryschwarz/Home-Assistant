"""HabitEngine — observes HA state changes and records habit events."""
from __future__ import annotations

import logging

from homeassistant.core import Event, HomeAssistant, callback

Class HabitEngine with:
- `__init__(self, hass: HomeAssistant, storage: AgentStorage, allowed_domains: list[str])`: store hass, storage, allowed_domains, _unsub_state=None, _cancel_purge=None
- `async_start(self)`:
  1. Run initial purge: `await self._storage.async_purge_old_events()`
  2. Subscribe: `self._unsub_state = self.hass.bus.async_listen("state_changed", self._handle_state_changed)`
  3. Schedule daily purge: `self._cancel_purge = async_track_time_interval(self.hass, self._async_daily_purge, timedelta(hours=24))`
- `async_stop(self)`:
  1. If _unsub_state: call it (synchronous callable), set None
  2. If _cancel_purge: call it (synchronous callable), set None
- `_handle_state_changed(self, event: Event)` (async):
  1. `new_state = event.data.get("new_state")` — return if None
  2. `old_state = event.data.get("old_state")` — return if None (entity creation, not a transition)
  3. Check `new_state.context.user_id is not None` — return if None (D-01: automation)
  4. `entity_id = new_state.entity_id`, `domain = entity_id.split(".")[0]`
  5. Check `domain in self._allowed_domains` — return if not (D-02)
  6. `service = self._infer_service(domain, old_state, new_state)`
  7. `persons_home = self._get_persons_home()`
  8. `weather_condition = self._get_weather_condition()`
  9. `await self._storage.async_record_event(entity_id, domain, service, persons_home, weather_condition)`
- `_infer_service(self, domain, old_state, new_state) -> str`:
  - If domain in ("light", "switch", "media_player"):
    - old="off", new="on" -> "turn_on"
    - old="on", new="off" -> "turn_off"
    - For media_player: old in ("paused","idle","off"), new="playing" -> "media_play"; new in ("paused","idle") -> "media_pause"
  - If domain == "climate":
    - If `new_state.attributes.get("temperature") != old_state.attributes.get("temperature")` -> "set_temperature"
    - old="off", new!="off" -> "turn_on"
    - old!="off", new="off" -> "turn_off"
  - Fallback: "state_change"
- `_get_persons_home(self) -> list[str] | None` (D-05):
  - `person_states = self.hass.states.async_all("person")`
  - If empty: return None
  - Return `[s.attributes.get("friendly_name", s.entity_id).lower() for s in person_states if s.state == "home"]`
- `_get_weather_condition(self) -> str | None` (D-06):
  - `weather_states = self.hass.states.async_all("weather")`
  - If empty: return None
  - Return `weather_states[0].state`
- `_async_daily_purge(self, _now=None)` (async callback for async_track_time_interval):
  - `await self._storage.async_purge_old_events()`
